public static class KupacElemzes
{
    public static int LevelekSzama<T>(Kupac<T> kupac)
    {
        int db = 0;

        for (int i = 0; i < kupac.n; i++)
        {
            if (Kupac<T>.Bal(i) >= kupac.n)
            {
                db++;
            }
        }

        return db;
    }
}
